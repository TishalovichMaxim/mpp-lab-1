using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using Moq;
using Tests.Input;

namespace Tests.Input.Tests
{
    [TestClass]
    public class ClassWithNotVoidReturnTypeMethodTests
    {
        private Mock<IDependency1> _dependency1;
        private Mock<IDependency2> _dependency2;
        private ClassWithNotVoidReturnTypeMethod _classWithNotVoidReturnTypeMethod;

        [TestInitialize]
        public void Initialization()
        {
            _dependency1 = new Mock<IDependency1>();
            _dependency2 = new Mock<IDependency2>();
            _classWithNotVoidReturnTypeMethod = new ClassWithNotVoidReturnTypeMethod(_dependency1.Object, _dependency2.Object);
        }

        [TestMethod]
        public void MyMethod()
        {
            string a = default(string);
            int b = default(int);

            string actual = _classWithNotVoidReturnTypeMethod.MyMethod(a, b);

            string expected = default(string);
            Assert.AreEqual(actual, expected);
            Assert.Fail("autogenerated");
        }
    }
}
